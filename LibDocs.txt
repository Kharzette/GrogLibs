MenuBuilder
	Quick lib for making quick menus
	Some sample code:

		//basic menu with some quickoptions
		mMenuBuilder	=new MenuBuilder(mFonts, mUITex);	//uitex made with texlibmaker
		mQOptions		=new QuickOptions(mMenuBuilder, mPlayerSteering,
			mScreenCenter, "SomeFont", "Textures\\UI\\hilite.png");

		//add a main menu screen
		mMenuBuilder.AddScreen("MainMenu", VerticalMenu,
			mScreenCenter, "Textures\\UI\\hilite.png");

		//some menu stops for the main menu
		mMenuBuilder.AddMenuStop("MainMenu", "Controls", "Controls", "Kawoszeh40");
		mMenuBuilder.AddMenuStop("MainMenu", "HowToPlay", "How to Play", "Kawoszeh40");
		mMenuBuilder.AddMenuStop("MainMenu", "Exit", "Exit", "Kawoszeh40");

		//a how to play screen
		mMenuBuilder.AddScreen("HowToPlay", MenuBuilder.ScreenTypes.InfoScreen, mScreenCenter, "");
		mMenuBuilder.SetBackGroundTexture("HowToPlay", "Textures\\UI\\FakeHowToPlay.png");
		mMenuBuilder.SetTimeOut("HowToPlay", 3000);

		//link stuff up
		mMenuBuilder.SetUpNav("MainMenu", "Controls");
		mMenuBuilder.Link("MainMenu", "Controls", "ControlsMenu");
		mMenuBuilder.Link("MainMenu", "HowToPlay", "HowToPlay");
		mMenuBuilder.Link("HowToPlay", "MainMenu");

		//watch for events in code
		eMenuStopInvoke	+=OnMenuStopInvoke;
		eNavigating		+=OnNavigating;
